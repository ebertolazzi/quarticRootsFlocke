1323 real_type x,

1337 // 3*x^2 + 2*a*x + b

1340 inline

1344 real_type a,
1345 real_type b,
1346 real_type c,
1347 real_type & p,
1348 real_type & dp,
1349 real_type & ddp
1350 ) {
